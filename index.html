Easy Deployment

Fork the Repo and use GitHub Actions

Paste this into your .github/workflows/node.js.yml file:

name: Node.js CI/CD Workflow

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
    - uses: actions/checkout@v3
    - uses: actions/setup-node@v3
      with:
        node-version: 20.x
    - run: npm install
    - run: npm start
